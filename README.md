<!-- 🚀 FEATURES -->
<h2>🚀 Features</h2>

<ul>
  <li>🔐 <b>User Authentication System</b>
    <ul>
      <li>Register new account</li>
      <li>Login system</li>
      <li>Password recovery</li>
    </ul>
  </li>
  <li>🎮 <b>Gameplay</b>
    <ul>
      <li>Movement using W / A / S / D</li>
      <li>Random food spawning</li>
      <li>Points & hearts system ❤️</li>
    </ul>
  </li>
  <li>🧠 <b>Game Logic</b>
    <ul>
      <li>Wall collision detection</li>
      <li>Score tracking</li>
      <li>Health system</li>
    </ul>
  </li>
</ul>
